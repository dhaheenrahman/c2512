    function NumOfDigits(number)
      set count = 0
      set copy = number
      while copy > 0
        copy = copy / 10
        count ++
      end while
      return count
    
    function ArmstrongOrNot(number)
      set sum = 0
      set copy = num
      pow = NumOfDigits(number)
      while copy > 0
        temp = copy % 10
        sum = sum + temp^pow
        copy = copy / 10
      end while
      if sum == number
        return true
      else
        return false
      end if

    pseudocode:
      input number
      if ArmstrongOrNot(number) == true
        print "Armstrong"
      else
        print "Not Armstrong"
      end if

function isArmstrong(number -> int) -> bool
	copyNumber = number
	int counter = 0
	while copyNumber > 0
		copyNumber = copyNumber/10
		counter = counter + 1
	end while
	copyNumber = number
	int sum = 0
	while copyNumber > 0
		temp = copyNumber % 10
		sum = sum + temp^counter
		copyNumber = copyNumber / 10
	end while
	if sum == number
		return true
	return false

function isArmstrongTest(){
	input number
	bool result = isArmStrong(number)
	if result == true
		print "Armstrong"
	else
		print "Not Armstrong"
	end if
}
